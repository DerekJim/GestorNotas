Algoritmo GestorDeNotas

    // Declaración de variables
    Definir cursos Como Cadena
    Definir notas Como Real
    Definir i, opcion, totalCursos Como Entero
    Definir suma, promedio Como Real
    

    // El maximo de cursos y notas que se pueden registrar
    Dimension cursos[6]   // máximo 6 cursos
    Dimension notas[6]    // máximo 6 notas

    // Inicialización
    totalCursos <- 0
    suma <- 0

    Repetir
        // Menú principal
        Imprimir "===== GESTOR DE NOTAS ====="
        Imprimir "1) Registrar curso y nota"
        Imprimir "2) Mostrar cursos y notas"
        Imprimir "3) Calcular promedio"
        Imprimir "4) Salir"
        Imprimir "Seleccione una opción:"
        Leer opcion

        Segun opcion Hacer

            1:  // Registrar curso y nota
                Si totalCursos < 6 Entonces
                    Imprimir "Ingrese el nombre del curso:"
                    Leer cursos[totalCursos+1]

                    // Validación de nota
                    Repetir
                        Imprimir "Ingrese la nota del curso (0 a 100):"
                        Leer notas[totalCursos+1]

                        Si EsNumero(notas[totalCursos+1]) Y notas[totalCursos+1] >= 0 Y notas[totalCursos+1] <= 100 Entonces
                            notaValida <- Verdadero
                        SiNo
                            Imprimir "Nota inválida. Debe ser un número entre 0 y 100."
                            notaValida <- Falso
                        FinSi
                    Hasta Que notaValida = Verdadero

                    // Sumar nota al total y actualizar contador
                    suma <- suma + notas[totalCursos+1]
                    totalCursos <- totalCursos + 1

                    Imprimir "Curso y nota registrados correctamente."

                SiNo
                    Imprimir "Ya alcanzó el máximo de cursos (6)."
                FinSi

            2:  // Mostrar cursos y notas
                Si totalCursos = 0 Entonces
                    Imprimir "No hay cursos registrados."
                SiNo
                    Imprimir "== LISTADO DE CURSOS Y NOTAS =="
                    Para i <- 1 Hasta totalCursos Hacer
                        Imprimir "Curso ", i, ": ", cursos[i], " | Nota: ", notas[i]
                    FinPara
                FinSi

            3:  // Calcular promedio
                Si totalCursos = 0 Entonces
                    Imprimir "No se puede calcular promedio: no hay cursos."
                SiNo
                    promedio <- suma / totalCursos
                    Imprimir "Promedio de ", totalCursos, " cursos: ", promedio
                FinSi

            4:  // Salir
                Imprimir "Saliendo..."

            De Otro Modo:
                Imprimir "Opción inválida."

        FinSegun

    Hasta Que opcion = 4

FinAlgoritmo
